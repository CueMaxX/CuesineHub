## Data formatting

CREATE TABLE `recipe`.`all` (
  `id` INT(100) NOT NULL AUTO_INCREMENT,
  `title` VARCHAR(100) NOT NULL,
  `steps` TEXT NOT NULL,
  `ingredients` VARCHAR(2048) NOT NULL,
  `difficulty` Int(1) NOT NULL,
  `picture` VARCHAR(100),
  `notes` TEXT,
  `time_minutes` Int(2) NOT NULL,
  `portions` Int(2) NOT NULL,
  `tags` VARCHAR(1024),
  PRIMARY KEY  (`id`)
);
